\begin{algorithm}
\caption{DLRA-CG scheme}
\label{alg:DLRA-CG}
\begin{algorithmic}[1]

\Require Hessian $H$, initial matrix $X^{(0)}$ with SVD $X^{(0)} = U_x \Sigma_x V_x^T$

\Ensure Low-rank solution $x = \operatorname{vec}(X)$

\State Compute initial gradient $G^{(0)} = \nabla \Phi(X^{(0)})$
\State Set initial search direction $D^{(0)} = -G^{(0)}$

\For{$i = 0,1,2,\dots,\text{max\_iter}$}

    \State \textbf{Step size:}
    \State $\displaystyle \alpha_i = \frac{\|G^{(i)}\|_F^2}{\langle D^{(i)}, H D^{(i)} \rangle_F}$
    \Comment{$HD^{(i)} = \operatorname{mat}[H \operatorname{vec}(D^{(i)})]$}

    \State \textbf{Dynamical low-rank update (KLS step):}
    \State $K' = U_x \Sigma_x + \alpha_i D^{(i)} V_x$
    \State $L' = V_x \Sigma_x + \alpha_i (D^{(i)})^T U_x$
    \State $\hat U = \operatorname{orth}([U_x, K'])$
    \State $\hat V = \operatorname{orth}([V_x, L'])$
    \State $\hat \Sigma = (\hat U^T U_x)\Sigma_x(V_x^T \hat V) + \alpha_i (\hat U^T D^{(i)} \hat V)$
    \State $(U_x, \Sigma_x, V_x) = \operatorname{truncate}(\hat U, \hat \Sigma, \hat V)$
    \Comment{$X^{(i+1)} = U_x \Sigma_x V_x^T$}

    \State \textbf{Search direction update:}
    \State $G^{(i+1)} = G^{(i)} + \alpha_i H D^{(i)}$ 
    \Comment{or $G^{(i+1)} = \nabla \Phi(X^{(i)})$}
    \State $\displaystyle \beta_i =
    \frac{\| G^{(i+1)} \|^2_F}
         {\| G^{(i)} \|^2_F}$
    \Comment{or $\displaystyle \beta_i =
    \frac{\langle G^{(i+1)}, H G^{(i)} \rangle_F}
         {\langle G^{(i)}, H G^{(i)} \rangle_F}$}
    \State $D^{(i+1)} = -G^{(i+1)} + \beta_i D^{(i)}$

\EndFor

\end{algorithmic}
\end{algorithm}

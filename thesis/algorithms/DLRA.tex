\begin{algorithm}
\caption{Schotthöfer et al.'s DLRA scheme}
\label{alg:dlra}
\begin{algorithmic}[1]

\Require Initial matrix $X^{(0)} = U_x \Sigma_x V_x^T$, solution rank $r$, step-size rule $\{\alpha_i\}$

\Ensure Rank-$r$ solution $x = \operatorname{vec}(X)$

\For{$i = 0,1,2,\dots,\text{max\_iter}$}

    \State \textbf{Search direction:}
    \State $D^{(i)} = -\nabla \Phi(X^{(i)})$

    \State \textbf{Dynamical low-rank update (KLS step):}
    \State $K' = U_x \Sigma_x + \alpha_i D^{(i)} V_x$
    \State $L' = V_x \Sigma_x + \alpha_i (D^{(i)})^T U_x$
    \State $\hat U = \operatorname{orth}([U_x, K'])$
    \State $\hat V = \operatorname{orth}([V_x, L'])$
    \State $\hat \Sigma = (\hat U^T U_x)\Sigma_x(V_x^T \hat V) + \alpha_i (\hat U^T D^{(i)} \hat V)$
    \State $(U_x, \Sigma_x, V_x) = \operatorname{truncate}_r(\hat U, \hat \Sigma, \hat V)$
    \Comment{$X^{(i+1)} = U_x \Sigma_x V_x^T$}

\EndFor

\end{algorithmic}
\end{algorithm}

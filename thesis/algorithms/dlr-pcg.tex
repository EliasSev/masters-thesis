\begin{algorithm}
\caption{Dynamical Low-Rank PCG Scheme}
\label{alg:dlr-pcg}
\begin{algorithmic}[1]

\Require Hessian $H$, initial matrix $X^{(0)}$ with SVD $X^{(0)} = U_x \Sigma_x V_x^T$, preconditioner $P^{-1}$

\Ensure Low-rank solution $x = \operatorname{vec}(X)$

\State Compute initial gradient $G^{(0)} = \nabla \Phi(X^{(0)})$
\State Precondition $Z^{(0)} = P^{-1} G^{(0)}$
\Comment $P^{-1} G^{(0)} =
\operatorname{mat}[P^{-1} \operatorname{vec}(G^{(0)})]$
\State Set initial search direction $D^{(0)} = -Z^{(0)}$

\For{$k = 0,1,2,\dots,\text{max\_iter}$}

    \State \textbf{Step size:}
    \State $\displaystyle \alpha_k =
    \frac{\langle G^{(k)}, Z^{(k)} \rangle_F}
         {\langle D^{(k)}, H D^{(k)} \rangle_F}$
    \Comment{$HD^{(k)} = \operatorname{mat}[H \operatorname{vec}(D^{(k)})]$}

    \State \textbf{Dynamical low-rank update:}
    \State $W' = U_x \Sigma_x + \alpha_k D^{(k)} V_x$
    \State $L' = V_x \Sigma_x + \alpha_k (D^{(k)})^T U_x$
    \State $\hat U = \operatorname{orth}([U_x, W'])$
    \State $\hat V = \operatorname{orth}([V_x, L'])$
    \State $\hat \Sigma = (\hat U^T U_x)\Sigma_x(V_x^T \hat V) + \alpha_k (\hat U^T D^{(k)} \hat V)$
    \State $(U_x, \Sigma_x, V_x) = \operatorname{truncate}(\hat U, \hat \Sigma, \hat V)$
    \Comment{$X^{(k)} = U_x \Sigma_x V_x^T$}

    \State \textbf{Search direction update:}
    \State $G^{(k+1)} = G^{(k)} + \alpha_k H D^{(k)}$
    \Comment{or $G^{(k+1)} = \nabla \Phi(X^{(k)})$ }
    \State $Z^{(k+1)} = P^{-1} G^{(k+1)}$
    \State $\displaystyle \beta_k =
    \frac{\langle G^{(k+1)}, Z^{(k+1)} \rangle_F}
         {\langle G^{(k)}, Z^{(k)} \rangle_F}$
    \Comment{or $\displaystyle \beta_k =
    \frac{\langle G^{(k+1)}, H Z^{(k)} \rangle_F}
         {\langle G^{(k)}, H Z^{(k)} \rangle_F}$}
    \State $D^{(k+1)} = -Z^{(k+1)} + \beta_k D^{(k)}$

\EndFor

\end{algorithmic}
\end{algorithm}

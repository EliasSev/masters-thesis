\section{Approximating the Forward Operator}\label{sec:approximating}
\input{figures/singular_values.tex}
% Opening: motivation and feasibility
The goal of this section is to compute the regularization weights given in Equation \eqref{eq:weights}, which requires the SVD $K = U\Sigma V^T$. Forming $K = TA^{-1}M$ requires solving $N$ linear systems $A v = M e_i$, and computing the SVD of the resulting dense $N_b \times N$ matrix costs $\mathcal{O}(N_b^2 N)$ \cite{dongarraSingularValueDecomposition2018}. In two spatial dimensions $N$ grows quadratically with mesh resolution, and in three dimensions cubically, making the construction of $K$ and finding its SVD infeasible.

The randomized SVD (Algorithm~\ref{alg:rsvd}) reduces this cost to $\mathcal{O}(N_b N l)$ \cite{halkoFindingStructureRandomness2011}, where $l = k + p$, $k$ is the target rank, and $p$ is an oversampling parameter (typically, $p \in [5, 10]$ \cite{halkoFindingStructureRandomness2011}). It yields a rank-$k$ approximation $K \approx U \Sigma V^T$, and since $K$ has a large nullspace and rapidly decaying singular values, as illustrated in Figure~\ref{fig:singular_values}, $k$ can be chosen much smaller than $N$.

The rSVD implementation of Algorithm~\ref{alg:rsvd} requires forming $B = Q^T K$, which requires $K$ to be stored as a dense $N_b \times N$ matrix. We instead propose a matrix-free variant without assembling $K$, at a cost of $l$ forward PDE solves and $l$ adjoint PDE solves.


\subsection{A matrix-free rSVD scheme}
\input{figures/rsvd.tex}
The algorithm consists of two stages. The first stage constructs an approximate basis $Q$ for the column space of $K$. We select a target rank $k$ and an oversampling parameter $p$, set $l = k + p$, and apply $K$ to $l$ random test vectors $\psi_1, \psi_2, \dots, \psi_l$ drawn from the standard normal distribution. This yields $l$ random samples $y_1, y_2, \dots, y_l$, where $y_i = K \psi_i$. Collecting these into matrices $Y = [y_i]_{i=1}^l$ and $\Psi = [\psi_i]_{i=1}^l$, we have
\begin{equation*}
    Y = K \Psi.
\end{equation*}
Each sample $K\psi_i$ is a linear combination of the columns of $K$, and tends to align with the dominant left singular vectors, see Figure~\ref{fig:rsvd}. Orthogonalizing the columns of $Y$ via a QR-factorization $QR = Y$ ensures that the $i$-th column $q_i$ of $Q$ approximately aligns with $u_i$, giving
\begin{equation*}
    \operatorname{span}(Q) \approx \operatorname{span}(K)
\end{equation*}
\nomenclature[G]{$Y$}{Sample matrix}
\nomenclature[G]{$\Psi$}{Random matrix}

The second stage projects $K$ onto $\operatorname{span}(Q)$ by computing $B = Q^T K$. Since we do not know $K$ explicitly, we use $K^*$ to compute $B$. In particular, by the definition of $K^*$,
\begin{equation}\label{eq:K_star}
    (K^* z)^T M = z^T M_\partial K.
\end{equation}
If we choose $z = M_{\partial}^{-1} q_i$ and insert it into Equation~\eqref{eq:K_star}, we see that
\begin{equation}\label{eq:rsvd_relation}
    (K^* z)^T M
    = (M_{\partial}^{-1} q_i)^T M_{\partial} K
    = q_i^T K.
\end{equation}
Setting $b_i = (K^* (M_{\partial}^{-1} q_i))^T M$ for $i = 1, 2, \dots, l$, and defining the $l \times N$ matrix
\begin{equation*}
    B = \begin{bmatrix} b_1 \\ \vdots \\ b_l \end{bmatrix},
\end{equation*}
we obtain, by Equation~\eqref{eq:rsvd_relation},
\begin{equation*}
    B = Q^T K.
\end{equation*}

Taking the SVD $B = \tilde U \Sigma V^T$, and setting $U = Q \tilde U$, gives an approximate SVD
\begin{equation}\label{eq:rsvd_approx}
    K \approx U \Sigma V^T.
\end{equation}

The factorization in \eqref{eq:rsvd_approx} is of rank $l$, since $U \Sigma V^T = Q \tilde U \Sigma V^T = Q B = Q Q^T K$ with $\operatorname{rank}(Q) = l$. A rank-$k$ factorization is obtained by selecting the $k$ first columns of the respective matrices in \eqref{eq:rsvd_approx}. The procedure is summarized in Algorithm~\ref{alg:discrete_rsvd}.

\subsection{Solver setup and precomputation}
\begin{itemize}
    \item Assembly of $M$ and $M_\partial$: $\mathcal{O}(N)$ and $\mathcal{O}(N_b)$
    \item LU factorization for $A$ (to apply $K$): $\mathcal{O}(N^{3/2})$
    \item LU factorization for $A$ (to apply $K^*$): $\mathcal{O}(N^{3/2})$
    \item Cholesky factorization for $M_\partial$ (for solving $M_\partial \tilde q_i = q_i$): $\mathcal{O}(N_b)$
    \item Cholesky factorization for $M$ (for solving $M \tilde q_i = q_i$): $\mathcal{O}(N^{3/2})$
\end{itemize}

Things to mention:
\begin{itemize}
    \item For large problems, use iterative Krylov solvers
    \item For self-adjoint problems, like here where $A = A^T$, only one LU-solver needs to be set up
    \item Once the factorization is computed, solving the system is equally fast
        \subitem LU: forward substitution $Ly = b$, then back substitution $Ux = y$ ($\mathcal{O}(N)$)
        \subitem Cholesky: forward substitution $Ly = b$, then back substitution $L^Tx = y$ ($\mathcal{O}(N)$)
    \item Note: $A$ is SPD here, can also use Cholesky
\end{itemize}

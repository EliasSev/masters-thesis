\section{Methods}

\subsection{Matrix-free randomized SVD}
\input{figures/rsvd.tex}
\textcolor{red}{Add oversampling parameter $p$.}
Let $K$ be the $N_b \times N$ discrete forward operator given in Eq.~\eqref{eq:K}. We are interested in a rank-$k$ approximation $K_k$ of the matrix $K$. By the Eckart–Young–Mirsky theorem, the optimal rank-$k$ approximation of $K$ is the truncated SVD obtained by keeping the $k$ first singular values, and their corresponding singular values. The randomized SVD provides an efficient alternative, by computing an approximate rank-$k$ factorization $K \approx U_k \Sigma_k V_k$, which also approximates the optimal truncated rank-$k$ SVD decomposition. We will now present a matrix-free variation of the rSVD algorithm, suited for problems where the matrix $K$ is not explicitly known.
\nomenclature[G]{$k$}{Target rank}

Like the rSVD, this matrix-free formulation consists of two stages. In the first stage, an approximate basis $Q$ of $K$ is constructed through random sampling of the column space of $K$. In the second stage, the matrix $K$ is projected onto the subspace $\operatorname{span}(Q)$, and a SVD factorization is computed of the projected matrix.

In the first stage, we apply $K$ on $k$ random vectors $\psi_1, \psi_2, \dots, \psi_k$ drawn from some distribution, typically the normal distribution. This yields the random samples $y_1, y_2, \dots, y_k$, where $y_i = K \psi_i$. If we gather the samples and random vectors into matrices $Y = [y_i]_{i=1}^k$ and $\Psi = [\psi_i]_{i}$, we can express $Y$ as
\begin{equation}
    Y = K \Psi.
\end{equation}
\textcolor{red}{(something about how $y_i$ is most likely to line up with the first singular vector)}. Geometrically, we want the $i$'th sample $y_i$ to approximately point in the direction of the the $i$ right singular vector $v_i$. To remove contributions of the previous right singular vectors $v_{i-1}, v_{i-2}, \dots, v_{1}$ on $y_i$, we must orthogonalize the random samples. A QR-factorization
\begin{equation}
    QR = Y
\end{equation}
achieves such an orthogonalization, obtaining the matrix $Q$ which contains the orthogonalized column-vectors $q_1, q_2, \dots, q_k$. This concludes the first step, where we have obtained an approximation of the column space of $K$:
\begin{equation}
    \operatorname{span}(Q) \approx \operatorname{span}(K)
\end{equation}
\nomenclature[G]{$Y$}{Sample matrix}
\nomenclature[G]{$\Psi$}{Random matrix}

For the second stage, we project $K$ down to the subspace $\operatorname{span}(Q)$. This is typically done by computing $B = Q^T K$. However, since we do not know $K$ explicitly, we instead compute $B^* = K^* Q$, and then $B$ can be found by computing $B = (B^*)^*$.

To do this in practice, we use the the definition of $K^*$ under the finite element space $V_h$,
\begin{equation}\label{eq:K_star}
    (K^* z)^T M = z^T M_\partial K.
\end{equation}
If we choose $z = M_{\partial}^{-1} q_i$ and insert it into Eq.~\eqref{eq:K_star}, we see that
\begin{equation}\label{eq:rsvd_relation}
    (K^* z)^T M
    = (M_{\partial}^{-1} q_i)^T M_{\partial} K
    = q_i^T K.
\end{equation}
Hence, if we let
\begin{equation}\label{eq:rsvd_adjoint_problem}
    b_i = (K^* (M_{\partial}^{-1} q_i))^T M
\end{equation}
for $i = 1, 2, \dots, k$, and define the $k \times N$ matrix
\begin{equation}
    B = \begin{bmatrix} b_1 \\ \vdots \\ b_k \end{bmatrix}
\end{equation}
it must be that, by Eq.~\eqref{eq:rsvd_relation},
\begin{equation}
    B = Q^T K,
\end{equation}
as desired.

We have computed the projection $B = Q^T K$ by solving the adjoint problem in Eq.~\eqref{eq:rsvd_adjoint_problem} $k$ times. In the final part of the second stage, we take the SVD of $B$,
\begin{equation}
    B = \tilde U_k \Sigma_k V_k^T,
\end{equation}
and set $U_k = Q \tilde U_k$. This gives an approximate SVD decomposition of the form
\begin{equation}\label{eq:rsvd_approx}
    K \approx U_k \Sigma_k V_k^T.
\end{equation}
By noting that $U_k \Sigma_k V_k^T = Q \tilde U_k \Sigma_k V_k^T = Q B = Q Q^T K$, and that $\operatorname{rank}(Q) = k$, we know that the factorization in Eq.~\eqref{eq:rsvd_approx} is a rank-$k$ approximation of $K$. The procedure is summarized in Algorithm~\ref{alg:discrete_rsvd}.


\begin{algorithm}[!thbp]
\caption{Discrete Operator rSVD Approximation of $K$}
\label{alg:discrete_rsvd}
\begin{algorithmic}[1]
    \Require Forward operator $K: \mathbb{R}^N \to \mathbb{R}^{N_b}$, adjoint operator $K^*: \mathbb{R}^{N_b} \to \mathbb{R}^N$, mass matrices $M, M_\partial$, target rank $k$, oversampling parameter $p$
    \Ensure Approximate rank-$k$ factorization $K \approx U \Sigma V^T$
    
    \State \textbf{Range sketch:}
    \State Set $l = k + p$
    \Comment Ensure that $l \le N_b$
    \For{$i = 1, \dots, l$}
        \State Draw a random vector $\psi_i \in \mathbb{R}^N$
        \State Apply the forward operator: $y_i = K \psi_i \in \mathbb{R}^{N_b}$
    \EndFor

    \State Form the matrix $Y = [y_1, \dots, y_l] \in \mathbb{R}^{N_b \times l}$ \Comment{Equivalent to $Y = K \Psi$}
    \State Compute an orthonormal basis: $Q = \text{orth}(Y)$

    \State \textbf{Projection onto the basis:}
    \For{$i = 1, \dots, l$}
        \State Let $q_i$ be the $i$-th column of $Q$
        \State Apply the adjoint operator: $\tilde b_i = K^* (M_{\partial}^{-1} q_i)$
        \State Set $b_i = \tilde b_i^T M$ \Comment{$i$-th row of $B$}
    \EndFor
    \State Form the projected matrix $B = \begin{bmatrix} b_1 \\ \vdots \\ b_l \end{bmatrix} \in \mathbb{R}^{l \times N}$ \Comment{Equivalent to $B = Q^T Y$}

    \State Compute the SVD: $B = \tilde U \Sigma V^T$
    \State Set $U = Q \tilde U$
    \State Truncate back to rank $k$: $U, \Sigma, V^T \leftarrow \operatorname{truncate}(U, \Sigma, V^T)$
    
\end{algorithmic}
\end{algorithm}

Let $K_k = U_k \Sigma_k V_k^T$ be the rank-$k$ approximate SVD of $K$ obtained using Algorithm~\ref{alg:discrete_rsvd}. The Tikhonov weights $W = \operatorname{diag}(w_1, \dots, N)$, given in Equation~\eqref{eq:weights_fem}, can then be approximated by setting
\begin{equation}\label{eq:rsvd_weights}
    y_i = V_k V_k^T e_i.
\end{equation}
To see this, let $K = U \Sigma V^T$ be the SVD of $K$ and we recall that $K^+ = V \Sigma^+ U^T$ (see Appendix~\textcolor{red}{(ref)}). Since $U_k$ is orthogonal, and since $\Sigma_k^+ \Sigma_k = I_k$ (\textcolor{red}{(need details)}), we have that $y = K_k^+ K_k e_i$ simplifies to Equation~\eqref{eq:rsvd_weights}.

\subsection{Dynamical low-rank solver}
Let $X = \operatorname{matrix}(x)$ and $X = U_x \Sigma_x V_x^T$ be the SVD of $X$ and let $\Phi(x)$ be some differentiable cost function of $x$, and $D = \operatorname{matrix}(\nabla \Phi(x))$ its matrix equivalent. We wish to update $X^{(k)}$ according to
\begin{equation}
    X^{(k+1)} = X^{(k)}-\alpha D
\end{equation}
Inserting the SVD of $X^{(k)}$ gives that
\begin{equation}\label{eq:gd_svd}
    U_x^{(k+1)} \Sigma_x^{(k+1)} (V_x^{(k+1)})^T = U_x^{(k)} \Sigma_x^{(k)} (V_x^{(k)})^T - \alpha D. 
\end{equation}
Right-multiplying by $V_x^{(k)}$ and denoting $W^{(k)} = U_x^{(k)} \Sigma_x^{(k)}$ yields
\begin{equation}
    W^{(k+1)}(V_x^{(k+1)})^T(V_x^{(k)}) = W^{(k)} - \alpha D V_x^{(k)}.
\end{equation}
The approximation $(V_x^{(k+1)})^T(V_x^{(k)}) \approx I$ suggests the following updating scheme for $W^{(k)}$:
\begin{equation}
    W^{(k+1)} = W^{(k)} - \alpha D V_x^{(k)}.
\end{equation}

To continue, transposing Equation~\eqref{eq:gd_svd}, right-multiply by $U_x^{(k)}$ and denoting $L^{(k)} = V_x^{(k)} (\Sigma_x^{(k)})^T$, we get a similar updating scheme for $L^{(k)}$:
\begin{equation}
    L^{(k+1)} = L^{(k)} - \alpha D^T U_x^{(k)}
\end{equation}

We now have update schemes for $W^{(k)} = U_x^{(k)} \Sigma_x^{(k)}$ and $L^{(k)} = V_x^{(k)} (\Sigma_x^{(k)})^T$. Further, we expand and orthogonalize the basis' $U^{(k)}$ and $V^{(k)}$ with the updated weighted basis' $W^{(k+1)}$ and $L^{(k+1)}$, respectively, in the following way:
\begin{equation}\label{eq:dynamic_orth}
    \hat U = \operatorname{orth}\left([
        U_x^{(k)} ,\, W^{(k+1)}
    ]\right) ,\,
    \hat V = \operatorname{orth}\left([
        V_x^{(k)} ,\, L^{(k+1)}
    ]\right).
\end{equation}

Currently, we have obtained an approximation $X \approx \hat U \Sigma_x \hat V^T$. In terms of $\Sigma_x$, we have $\Sigma_x \approx \hat U^T X \hat V$. Substituting $X = U_x \Sigma_x V_x^T$ gives a approximation $\hat \Sigma^{(k)}$ of $\Sigma_x$,
\begin{equation}\label{eq:sigma_approx}
    \Sigma_x \approx \hat \Sigma^{(k)} =(\hat U^T U_x)\Sigma_x(V_x^T \hat V)
\end{equation}
Now, if we write the gradient descent scheme $X^{(k+1)} = X^{(k)} - \alpha D$ as
\begin{equation}
    \hat U^T X^{(k+1)} \hat V = \hat U^TX^{(k)}\hat V - \alpha \hat U^T D \hat V,
\end{equation}
and use Equation~\eqref{eq:sigma_approx}, we get the following updating scheme for $\hat \Sigma^{(k)}$:
\begin{eqnarray}
    \hat \Sigma^{(k+1)} = \hat \Sigma^{(k)} - \alpha \hat U^T D \hat V.
\end{eqnarray}

As a consequence of Equation~\eqref{eq:dynamic_orth}, the rank of $\hat U$, $\hat S$, and $\hat V$ doubles. Therefore, the matrices need to be truncated back to a low-rank profile,
\begin{equation}
    (U_x^{(k+1)}, \Sigma_x^{(k+1)}, V_x^{(k+1)}) = \operatorname{truncate}(\hat U, \hat \Sigma, \hat V).
\end{equation}

Description of the dynamical low-rank CG method. I am using the cost function
\begin{equation*}
    \Phi(x) = \frac 1 2 \|Kx - y\|_{M_\partial}^2 + \frac \lambda 2 \| W x \|_M^2,
\end{equation*}
with gradient
\begin{equation*}
    \nabla \Phi (x) = K^T M_\partial (K x - y) + \lambda W^T M W x,
\end{equation*}
and Hessian
\begin{equation*}
    \nabla^2 \Phi = K^T M_\partial K + \lambda W^T M W.
\end{equation*}
Here, $K$ is the forward operator, $W$ is the weight matrix, $M$ is the mass matrix, and $M_\partial$ is the boundary mass matrix.

\subsection{Dynamical Low-Rank Preconditioned CG Scheme}
Preconditioning is a standard technique for accelerating CG methods. The system $H x = y$ is replaced by the equivalent system $P^{-1} H x = P^{-1} b$, where $P \approx H$ is the preconditioner matrix. The convergence of CG depends on $\kappa(H)$ (see Sec.~6.11.3 in Saad) \cite{saadIterativeMethodsSparse2003}, so $P$ is chosen such that $\kappa(P^{-1} H) \ll \kappa(H)$. For elliptic PDE-based forward operators $K$, the Hessian $H = K^T M_\partial K + \lambda W^T M W$ typically has a large condition number, making preconditioning effective.

In the dynamical low-rank CG scheme, the preconditioned system $P^{-1} H x = P^{-1} b$ leads to the following adjustments in Algorithm \ref{alg:dlr-cg}:
\begin{equation*}
    \alpha_k = \frac
    {\langle G_k, Z_k \rangle_F}
    {\langle D_k, H D_k \rangle_F}, \qquad
    \beta_k = \frac
    {\langle G_{k+1}, Z_{k+1} \rangle_F}
    {\langle G_k, Z_k \rangle_F}, \qquad
    D_{k+1} = -Z_{k+1} + \beta_k D_k
\end{equation*}
These modifications are equivalent with standard preconditioned conjugate gradient (PCG), as originally described by Concus et al. \cite{concusGeneralizedConjugateGradient1976}. The resulting dynamical low-rank PCG scheme is summarized in Algorithm \ref{alg:dlr-pcg}.

\subsubsection{Choice of Preconditioner}
The effectiveness of PCG depends on the choice of $P$. A good preconditioner must approximate $H$ well, but must be cheap to apply at each iteration. We consider three common preconditioners: Jacobi scaling, incomplete Cholesky, and (sparse) Cholesky. We consider three standard preconditioners: Jacobi scaling, incomplete Cholesky (IC), and (sparse) Cholesky. Since $H =  K^T M_\partial K + \lambda W^T M W$ has a sparse plus low-rank structure, the Cholesky-based preconditioners are modified using the Woodbury formula to exploit this \cite{woodbury1950inverting}. We refer to Golub and Van Loan for a comprehensive treatment of preconditioning and iterative methods \cite{golubMatrixComputations4th2013}.

\paragraph{Jacobi scaling.} A Jacobi preconditioner $P = \operatorname{diag}(H)$ approximates $H$ by its diagonal. The approximation is cheap to compute, as $P^{-1}$ is the inverse of a diagonal matrix. It is most effective when $H$ is diagonally dominant.
 
\paragraph{Incomplete Cholesky.} For a symmetric positive definite (SPD) matrix $S$, the incomplete Cholesky factorization computes an approximation $S \approx  L L^T$, where $L$ is lower triangular. The Hessian $H = K^T M_\partial K + \lambda W^T M W$ consists of a low-rank part plus a sparse part. We only apply the IC factorization to the sparse part $S = \lambda W^T M W$, yielding $S \approx L L^T$. To account for the low-rank part, we let $K = U \Sigma V^T$ be the SVD of $K$ (which is assumed to be of rank $k$), and define $F = V \Sigma L_U$, where $L_U$ is the Cholesky factor of the $k \times k$ matrix $U^T M_\partial U$ (which is SPD since $M_\partial$ is SPD and $U$ has linearly independent columns). Then, the low-rank part can be written as $F F^T = V \Sigma L_U L_U^T \Sigma V^T = K^T M_\partial K$. By the Woodbury formula,
\begin{equation*}
    (S + FF^T)^{-1} = S^{-1} - 
    S^{-1} F (I_k + F^T S^{-1} F)^{-1} F^T S^{-1}.
\end{equation*}
Replacing $S^{-1}$ with the approximate inverse $L^{-T} L^{-1}$ yields the preconditioner
\begin{equation*}
    P^{-1} = L^{-T} L^{-1} -
    L^{-T} (L^{-1} F) C^{-1} (L^{-1} F)^T L^{-1},
\end{equation*}
where $C = I_k + (L^{-1} F)^T(L^{-1} F)$ is a $k \times k$ SPD matrix.

\paragraph{Sparse Cholesky.} The sparse Cholesky preconditioner follows the same Woodbury construction as the incomplete Cholesky preconditioner, but replaces the approximate factorization of $S$ with the exact sparse Cholesky factorization $S = LL^T$. Since the Woodbury formula then gives $P^{-1} = H^{-1}$, this is equivalent to solving $H x = y$ directly, and CG converges in a single iteration. However, the setup is more expensive compared to incomplete Cholesky.
